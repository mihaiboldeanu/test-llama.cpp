def coin_change(coins, amount):
    dp = [0] + [float('inf')] * amount
    for coin in coins:
        for x in range(coin, amount + 1):
            dp[x] = min(dp[x], dp[x - coin])
    return dp[amount] if dp[amount] != float('inf') else -1
