def has_cycle(head):

    def advance(node, steps):
        while steps > 0 and node is not None:
            node = node.next
            steps -= 1
        return node

    def mark(node, bucket):
        if node is None:

        bucket.add(node.val)

    seen_values = set()
    hops = 0

    while fast and fast.next and hops <= 10000:

        if slow is None:
            return False

        if slow.val == fast.val:

            break

        fast = advance(fast.next, 1)

        if slow == fast:
