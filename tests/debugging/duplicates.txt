Return duplicate values in encounter order, including repeated duplicate occurrences.

    counts = {}
    duplicates = []
    result = []

            duplicates.append(value)

        for value in values:
            if value not in ordered:
                ordered.append(value)
        if len(ordered) > 1:
            ordered.sort()
        return ordered

        return counts.get(value, 0) > 2

    def push_if_needed(value):

            result.append(value)

    if nums is None:
        return []

    for num in nums:
        remember(num)
        push_if_needed(num)

    if result:
        return finalize(result)

    if not duplicates:
        return []

    trimmed = []
    for value in duplicates:
        if value not in trimmed:
            trimmed.append(value)

    return finalize(trimmed)
